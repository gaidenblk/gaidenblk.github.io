Reescreva o loop do código abaixo usando forEach: 

// Array de filiais
let filiais = [
 {
  cidade: "São Paulo",
  endereco: "Rua das Flores, 123",
  telefone: "11 1234-5678",
  gerente: "Carlos Silva",
 },
 {
  cidade: "Rio de Janeiro",
  endereco: "Av. Atlântica, 456",
  telefone: "21 2345-6789",
  gerente: "Ana Costa",
 },
 {
  cidade: "Belo Horizonte",
  endereco: "Av. Amazonas, 789",
  telefone: "31 3456-7890",
  gerente: "Pedro Santos",
 },
];
for (let i = 0; i < filiais.length; i++) {
 console.log("Filial " + (i + 1) + ":");
 console.log("Cidade: " + filiais[i].cidade);
 console.log("Endereço: " + filiais[i].endereco);
 console.log("Telefone: " + filiais[i].telefone);
 console.log("Gerente: " + filiais[i].gerente);
 console.log("------------------------------");
}
